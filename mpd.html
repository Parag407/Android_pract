Practical 1: Creating a Simple App

Steps

Practical 2: Lifecycle of an Android Activity

Steps

  1. Right-click in the `MainActivity.java` file after the `onCreate()` method.
  2. Select "Generate" and then "Override Methods".
  3. Select the methods of the Android Lifecycle Activity (`onStart`, `onStop`, `onPause`, etc.) and click "OK" to proceed.
  4. After the methods are added, add a `Toast` message to each of them to show when the method is called.

File: MainActivity.java

package com.example.myapplication;

import androidx.appcompat.app.AppCompatActivity;
import android.os.Bundle;
import android.widget.Toast;

public class MainActivity extends AppCompatActivity {

        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);
        Toast.makeText(this, "onCreate() method called", Toast.LENGTH_SHORT).show();
    }

    @Override
    protected void onStart() {
        super.onStart();
        Toast.makeText(this, "onStart() method called", Toast.LENGTH_SHORT).show();

File: MainActivity.java

package com.example.myapplication;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText _username, _password;
    Button _btnSubmitLogin;
    String username, password;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        _username = findViewById(R.id.main_et_username);
        _password = findViewById(R.id.main_et_password);
        _btnSubmitLogin = findViewById(R.id.main_btn_login);

        _btnSubmitLogin.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                username = _username.getText().toString().trim();
                password = _password.getText().toString().trim();

                if (username.isEmpty()) {
                    _username.setError("Please enter Username");
                    _username.requestFocus();
                } else if (password.isEmpty()) {
                    _password.setError("Password cannot be blank");
                    _password.requestFocus();
                } else if (password.length() < 8) {
                    _password.setError("Password should be greater than 8 characters");
                    _password.requestFocus();
                } else {
                    if (username.equals("admin") && password.equals("password@1234")) {
                        Toast.makeText(MainActivity.this, "Login Successful", Toast.LENGTH_SHORT).show();
                    } else {
                        Toast.makeText(MainActivity.this, "Failed to Login, Invalid Credentials", Toast.LENGTH_SHORT).show();
                    }
                }
            }
        });
    }
}

Practical 5: Go from One Activity to Another

Steps

  1. Create a new `Empty Views Activity` by right-clicking on the Java folder, and selecting `New > Activity > Empty Views Activity`. Name it `SecondActivity`.
  2. Add an `Intent` code to `MainActivity.java` to navigate to `SecondActivity` upon successful login. Use `putExtra` to pass the username and a login status.
  3. In `activity_second.xml`, create a `LinearLayout` with a `TextView` to display the Dashboard heading and another to display the logged-in username.
  4. In `SecondActivity.java`, retrieve the `Intent` data from `MainActivity` using `getIntent()` and `getExtras()`. Then set the text of the `TextView` with the retrieved username.

File: MainActivity.java (Updated)

package com.example.myapplication;

import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText _username, _password;
    Button _btnSubmitLogin;
    String username, password;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        _username = findViewById(R.id.main_et_username);
        _password = findViewById(R.id.main_et_password);
        _btnSubmitLogin = findViewById(R.id.main_btn_login);

        _btnSubmitLogin.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                username = _username.getText().toString().trim();
                password = _password.getText().toString().trim();

                if (username.isEmpty()) {
                    _username.setError("Please enter Username");
                    _username.requestFocus();
                } else if (password.isEmpty()) {
                    _password.setError("Password cannot be blank");
                    _password.requestFocus();
                } else if (password.length() < 8) {
                    _password.setError("Password should be greater than 8 characters");
                    _password.requestFocus();
                } else {
                    if (username.equals("admin") && password.equals("password@1234")) {
                        Intent i = new Intent(MainActivity.this, SecondActivity.class);
                        i.putExtra("username", username);
                        i.putExtra("isLoggedIn", "1");
                        startActivity(i);
                        finish();
                        Toast.makeText(MainActivity.this, "Login Successful", Toast.LENGTH_SHORT).show();
                    } else {
                        Toast.makeText(MainActivity.this, "Failed to Login, Invalid Credentials", Toast.LENGTH_SHORT).show();
                    }
                }
            }
        });
    }
}

package com.example.myapplication;

import android.content.Intent;
import android.os.Bundle;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class SecondActivity extends AppCompatActivity {
    TextView _second_tv_username;
    String username;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        _second_tv_username = findViewById(R.id.second_tv_username);

        Intent i = getIntent();
        Bundle b = i.getExtras();

        if (b != null && b.containsKey("isLoggedIn")) {
            if (b.getString("isLoggedIn").equals("1")) {
                username = b.getString("username", "");
                _second_tv_username.setText("Welcome, " + username);
            }
        }
    }
}

package com.example.myapplication;

import android.content.Intent;
import android.os.Bundle;
import android.view.Menu;
import android.view.MenuInflater;
import android.view.MenuItem;
import android.view.View;
import android.widget.ImageView;
import android.widget.TextView;
import android.widget.Toast;
import androidx.annotation.NonNull;
import androidx.appcompat.app.AppCompatActivity;

public class SecondActivity extends AppCompatActivity {
    TextView _second_tv_username;
    String username;
    ImageView _second_iv_image;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        _second_tv_username = findViewById(R.id.second_tv_username);
        _second_iv_image = findViewById(R.id.second_iv_image);

        Intent i = getIntent();
        Bundle b = i.getExtras();

        if (b != null && b.containsKey("isLoggedIn")) {
            if (b.getString("isLoggedIn").equals("1")) {
                username = b.getString("username", "");
                _second_tv_username.setText("Welcome, " + username);
            }
        }

        _second_iv_image.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                _second_iv_image.setImageResource(R.drawable.sunrise);
            }
        });
    }

    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        MenuInflater inflater = getMenuInflater();
        inflater.inflate(R.menu.basic_menu, menu);
        return true;
    }

    @Override
    public boolean onOptionsItemSelected(@NonNull MenuItem item) {
        int id = item.getItemId();
        if (id == R.id.menu_home) {
            Toast.makeText(this, "Home Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_about) {
            Toast.makeText(this, "About Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_contact) {
            Toast.makeText(this, "Contact Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_gallery) {
            Toast.makeText(this, "Gallery Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_logout) {
            Toast.makeText(this, "Logout Menu Item Clicked", Toast.LENGTH_SHORT).show();
        }
        return true;
    }
}

Practical 9: Implement use of SharedPreferences

Steps

  1. In `MainActivity.java`, declare a `SharedPreferences` object. Check if a username is already saved in `SharedPreferences`. If it is, redirect the user to `SecondActivity` to maintain a logged-in session.
  2. When the user logs in, remove the `Intent` code and save the username to `SharedPreferences` using an `Editor`.
  3. In `SecondActivity.java`, retrieve the username from `SharedPreferences` to display it on the dashboard.
  4. Implement the logout functionality. When the "Logout" menu item is clicked, clear the `SharedPreferences` and redirect the user back to `MainActivity`.

File: MainActivity.java (Updated)

package com.example.myapplication;

import android.content.Context;
import android.content.Intent;
import android.content.SharedPreferences;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.EditText;
import android.widget.Toast;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {
    EditText _username, _password;
    Button _btnSubmitLogin;
    String username, password;
    SharedPreferences sharedPreferences;
    public static final String myPrefs = "myPrefs";

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        sharedPreferences = getSharedPreferences(myPrefs, Context.MODE_PRIVATE);
        if (sharedPreferences.contains("username")) {
            Intent i = new Intent(MainActivity.this, SecondActivity.class);
            startActivity(i);
            finish();
        }

        _username = findViewById(R.id.main_et_username);
        _password = findViewById(R.id.main_et_password);
        _btnSubmitLogin = findViewById(R.id.main_btn_login);

        _btnSubmitLogin.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                username = _username.getText().toString().trim();
                password = _password.getText().toString().trim();

                if (username.isEmpty()) {
                    _username.setError("Please enter Username");
                    _username.requestFocus();
                } else if (password.isEmpty()) {
                    _password.setError("Password cannot be blank");
                    _password.requestFocus();
                } else if (password.length() < 8) {
                    _password.setError("Password should be greater than 8 characters");
                    _password.requestFocus();
                } else {
                    if (username.equals("admin") && password.equals("password@1234")) {
                        SharedPreferences.Editor editor = sharedPreferences.edit();
                        editor.putString("username", username);
                        editor.apply();
                        
                        Intent i = new Intent(MainActivity.this, SecondActivity.class);
                        startActivity(i);
                        finish();
                        Toast.makeText(MainActivity.this, "Login Successful", Toast.LENGTH_SHORT).show();
                    } else {
                        Toast.makeText(MainActivity.this, "Failed to Login, Invalid Credentials", Toast.LENGTH_SHORT).show();
                    }
                }
            }
        });
    }

package com.example.myapplication;

public class SecondActivity extends AppCompatActivity {
    TextView _second_tv_username;
    String username;
    ImageView _second_iv_image;
    SharedPreferences sharedPreferences;
    public static final String myPrefs = "myPrefs";

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_second);

        _second_tv_username = findViewById(R.id.second_tv_username);
        _second_iv_image = findViewById(R.id.second_iv_image);
        sharedPreferences = getSharedPreferences(myPrefs, Context.MODE_PRIVATE);

        if (sharedPreferences.contains("username")) {
            username = sharedPreferences.getString("username", "");
            if (!username.isEmpty()) {
                _second_tv_username.setText("Welcome, " + username);
            }
        } else {
            // No saved session, redirect to login
            Intent i = new Intent(SecondActivity.this, MainActivity.class);
            startActivity(i);
            finish();
        }

        _second_iv_image.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View view) {
                _second_iv_image.setImageResource(R.drawable.sunrise);
            }
        });
    }
    
    @Override
    public boolean onCreateOptionsMenu(Menu menu) {
        MenuInflater inflater = getMenuInflater();
        inflater.inflate(R.menu.basic_menu, menu);
        return true;
    }

    @Override
    public boolean onOptionsItemSelected(@NonNull MenuItem item) {
        int id = item.getItemId();
        if (id == R.id.menu_home) {
            Toast.makeText(this, "Home Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_about) {
            Toast.makeText(this, "About Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_contact) {
            Toast.makeText(this, "Contact Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_gallery) {
            Toast.makeText(this, "Gallery Menu Item Clicked", Toast.LENGTH_SHORT).show();
        } else if (id == R.id.menu_logout) {
            SharedPreferences.Editor editor = sharedPreferences.edit();
            editor.remove("username");
            editor.apply();
            
            Intent i = new Intent(SecondActivity.this, MainActivity.class);
            startActivity(i);
            finish();
            Toast.makeText(this, "Logout Menu Item Clicked", Toast.LENGTH_SHORT).show();
        }
        return true;
    }
}

Practical 10: Different Orientation Screens

Steps

  1. In `AndroidManifest.xml`, add the `android:screenOrientation` attribute to an `activity` to lock its orientation.
  2. In Android Studio, you can create a landscape-specific layout file. Navigate to `activity_main.xml` and click on the file name in the preview pane. Select `Create Landscape Qualifier`.
  3. In the newly created `activity_main.xml (land)` file, modify the layout to have a different design for landscape mode, for example, changing the "Login" text to "Register".
